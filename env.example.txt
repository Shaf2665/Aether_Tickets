# Aether Tickets Configuration
# Copy this content to a file named .env and fill in your values

# ============================================================================
# DISCORD BOT CONFIGURATION
# ============================================================================

# Required: Your Discord bot token
# Get this from https://discord.com/developers/applications
DISCORD_BOT_TOKEN=your_bot_token_here

# Optional: Guild (server) ID for faster command syncing during development
# Leave empty for global command syncing
GUILD_ID=

# Optional: Category ID where tickets will be created
# Leave empty to create tickets in the root of the server
TICKET_CATEGORY_ID=

# Optional: Support role ID for staff access to tickets
# Leave empty if you don't want to grant a role access to all tickets
SUPPORT_ROLE_ID=

# Optional: Channel ID where the ticket panel will be sent
# Leave empty if you don't want an automatic ticket panel
TICKET_CHANNEL_ID=

# ============================================================================
# FLASK WEB UI CONFIGURATION
# ============================================================================

# Required: Discord OAuth Client ID
# Create an OAuth2 application at https://discord.com/developers/applications
DISCORD_CLIENT_ID=your_client_id_here

# Required: Discord OAuth Client Secret
DISCORD_CLIENT_SECRET=your_client_secret_here

# Required: Flask secret key for session management
# Generate a random key: python -c "import secrets; print(secrets.token_hex(32))"
FLASK_SECRET_KEY=your_secret_key_here

# Optional: Flask environment (development or production)
# Set to 'development' for debug mode, 'production' for normal use
FLASK_ENV=development

# Optional: Discord OAuth redirect URI
# Change this if running on a different host/port
DISCORD_REDIRECT_URI=http://localhost:5000/auth/callback

# Optional: Database path
# Leave empty to use tickets.db in the current directory
DATABASE_PATH=tickets.db
